25 inline void ImpulseDynamicsForwardEuler::linearizeImpulseDynamics(
26  Robot& robot, const ImpulseStatus& impulse_status,
27  const ImpulseSplitSolution& s, ImpulseSplitKKTMatrix& kkt_matrix,
28  ImpulseSplitKKTResidual& kkt_residual) {
29  setImpulseStatus(impulse_status);
30  linearizeInverseImpulseDynamics(robot, impulse_status, s, data_);
31  linearizeImpulseVelocityConstraint(robot, impulse_status, data_);
32  // augment inverse impulse dynamics constraint
33  kkt_residual.lq().noalias() += data_.dImDdq().transpose() * s.beta;
34  kkt_residual.ldv.noalias() += data_.dImDddv.transpose() * s.beta;
35  // We use an equivalence dmIDdf_().transpose() = - dCdv_() = - dCddv, to avoid
36  // redundant calculation of dImDdf_().
37  kkt_residual.lf().noalias() -= data_.dCdv() * s.beta;
38  // augment impulse velocity constraint
39  kkt_residual.lq().noalias() += data_.dCdq().transpose() * s.mu_stack();
40  kkt_residual.lv().noalias() += data_.dCdv().transpose() * s.mu_stack();
41  // We use an equivalence dCdv_() = dCddv, to avoid redundant calculation.
42  kkt_residual.ldv.noalias() += data_.dCdv().transpose() * s.mu_stack();
43 }

72 inline void ImpulseDynamicsForwardEuler::condensing(
73  const Robot& robot, const ImpulseStatus& impulse_status,
74  ImpulseDynamicsForwardEulerData& data, ImpulseSplitKKTMatrix& kkt_matrix,
75  ImpulseSplitKKTResidual& kkt_residual) {
76  const int dimv = robot.dimv();
77  const int dimf = impulse_status.dimf();
78  data.MJtJinv_dImDCdqv().leftCols(dimv).noalias()
79  = data.MJtJinv() * data.dImDCdq();
80  data.MJtJinv_dImDCdqv().topRightCorner(dimv, dimv).noalias()
81  = data.MJtJinv().topRightCorner(dimv, dimf) * data.dCdv();
82  data.MJtJinv_dImDCdqv().bottomRightCorner(dimf, dimv).noalias()
83  = data.MJtJinv().bottomRightCorner(dimf, dimf) * data.dCdv();
84  data.MJtJinv_ImDC().noalias() = data.MJtJinv() * data.ImDC();
85  data.Qdvfqv().topRows(dimv).noalias()
86  = (- kkt_matrix.Qdvdv().diagonal()).asDiagonal()
87  * data.MJtJinv_dImDCdqv().topRows(dimv);
88  data.Qdvfqv().bottomRows(dimf).noalias()
89  = - kkt_matrix.Qff() * data.MJtJinv_dImDCdqv().bottomRows(dimf);
90  data.ldv() = kkt_residual.ldv;
91  data.lf() = - kkt_residual.lf();
92  data.ldv().noalias()
93  -= kkt_matrix.Qdvdv().diagonal().asDiagonal()
94  * data.MJtJinv_ImDC().head(dimv);
95  data.lf().noalias() -= kkt_matrix.Qff() * data.MJtJinv_ImDC().tail(dimf);
96  kkt_matrix.Qxx().noalias()
97  -= data.MJtJinv_dImDCdqv().transpose() * data.Qdvfqv();
98  kkt_residual.lx().noalias()
99  -= data.MJtJinv_dImDCdqv().transpose() * data.ldvf();
100  kkt_matrix.Fvq() = - data.MJtJinv_dImDCdqv().topLeftCorner(dimv, dimv);
101  kkt_matrix.Fvv() = Eigen::MatrixXd::Identity(dimv, dimv)
102  - data.MJtJinv_dImDCdqv().topRightCorner(dimv, dimv);
103  kkt_residual.Fv().noalias() -= data.MJtJinv_ImDC().head(dimv);
104 }
