215 inline void SplitSolution::integrate(const Robot& robot, const double step_size,
216  const SplitDirection& d) {
217  lmd.noalias() += step_size * d.dlmd();
218  gmm.noalias() += step_size * d.dgmm();
219  robot.integrateConfiguration(d.dq(), step_size, q);
220  v.noalias() += step_size * d.dv();
221  a.noalias() += step_size * d.da();
222  u.noalias() += step_size * d.du();
223  beta.noalias() += step_size * d.dbeta();
224  if (has_floating_base_) {
225  nu_passive.noalias() += step_size * d.dnu_passive;
226  }
227  if (has_active_contacts_) {
228  assert(f_stack().size() == d.df().size());
229  f_stack().noalias() += step_size * d.df();
230  set_f_vector();
231  assert(mu_stack().size() == d.dmu().size());
232  mu_stack().noalias() += step_size * d.dmu();
233  set_mu_vector();
234  }
235  if (has_active_impulse_) {
236  assert(xi_stack().size() == d.dxi().size());
237  xi_stack().noalias() += step_size * d.dxi();
238  }
239 }
