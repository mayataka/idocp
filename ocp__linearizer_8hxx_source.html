138  kkt_matrix[i], kkt_residual[i]);
139  }

159  }

161  Algorithm::run(

163  contact_sequence.contactStatus(ocp.discrete().contactPhase(i)),

168  }

171  ocp.discrete().t(N), q_prev(ocp.discrete(), q, s, N), s[N],
172  kkt_matrix[N], kkt_residual[N]);
173  }

180  ocp.discrete().t_impulse(impulse_index),
181  s[time_stage_before_impulse].q, s.impulse[impulse_index],
182  s.aux[impulse_index], kkt_matrix.impulse[impulse_index],
183  kkt_residual.impulse[impulse_index]);
184  }

189  Algorithm::run(
190  ocp.aux[impulse_index], robots[omp_get_thread_num()],
191  contact_sequence.contactStatus(
192  ocp.discrete().contactPhaseAfterImpulse(impulse_index)),
193  ocp.discrete().t_impulse(impulse_index),
194  ocp.discrete().dt_aux(impulse_index), s.impulse[impulse_index].q,
195  s.aux[impulse_index], s[time_stage_after_impulse],
196  kkt_matrix.aux[impulse_index], kkt_residual.aux[impulse_index]);
197  }
198  else {

200  const int time_stage_after_lift

202  if (ocp.discrete().isTimeStageBeforeImpulse(time_stage_after_lift)) {

215  }

231 inline const Eigen::VectorXd& OCPLinearizer::q_prev(
232  const OCPDiscretizer& discretizer, const Eigen::VectorXd& q,
233  const Solution& s, const int time_stage) {
234  assert(time_stage >= 0);
235  assert(time_stage <= discretizer.N());
236  if (time_stage == 0) {
237  return q;
238  }
239  else if (discretizer.isTimeStageBeforeImpulse(time_stage-1)) {
240  return s.aux[discretizer.impulseIndexAfterTimeStage(time_stage-1)].q;
241  }
242  else if (discretizer.isTimeStageBeforeLift(time_stage-1)) {
243  return s.lift[discretizer.liftIndexAfterTimeStage(time_stage-1)].q;
244  }
245  else {
246  return s[time_stage-1].q;
247  }
