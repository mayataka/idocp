48 inline void ContactDynamics::linearizeContactDynamics(
49  Robot& robot, const ContactStatus& contact_status, const double dt,
50  const SplitSolution& s, SplitKKTResidual& kkt_residual) {
51  assert(dt > 0);
52  setContactStatus(contact_status);
53  linearizeInverseDynamics(robot, contact_status, s, data_);
54  linearizeContactConstraint(robot, contact_status, baumgarte_time_step_, data_);
55  // augment inverse dynamics constraint
56  kkt_residual.lq().noalias() += dt * data_.dIDdq().transpose() * s.beta;
57  kkt_residual.lv().noalias() += dt * data_.dIDdv().transpose() * s.beta;
58  kkt_residual.la.noalias() += dt * data_.dIDda.transpose() * s.beta;
59  if (has_active_contacts_) {
60  // We use an equivalence dIDdf_().transpose() = - dCda_(), to avoid
61  // redundant calculation of dIDdf_().
62  kkt_residual.lf().noalias() -= dt * data_.dCda() * s.beta;
63  }
64  // augment floating base constraint
65  if (has_floating_base_) {
66  kkt_residual.lu_passive = dt * s.nu_passive;
67  kkt_residual.lu_passive.noalias() -= dt * s.beta.template head<kDimFloatingBase>();
68  kkt_residual.lu().noalias() -= dt * s.beta.tail(robot.dimu());
69  }
70  else {
71  kkt_residual.lu().noalias() -= dt * s.beta;
72  }
73  // augment contact constraint
74  if (has_active_contacts_) {
75  kkt_residual.lq().noalias()
76  += dt * data_.dCdq().transpose() * s.mu_stack();
77  kkt_residual.lv().noalias()
78  += dt * data_.dCdv().transpose() * s.mu_stack();
79  kkt_residual.la.noalias() += dt * data_.dCda().transpose() * s.mu_stack();
80  }
81 }

105 inline void ContactDynamics::condenseContactDynamics(
106  Robot& robot, const ContactStatus& contact_status, const double dt,
107  SplitKKTMatrix& kkt_matrix, SplitKKTResidual& kkt_residual,
108  const bool is_forward_euler) {
109  assert(dt > 0);
110  const int dimf = contact_status.dimf();
111  robot.computeMJtJinv(data_.dIDda, data_.dCda(), data_.MJtJinv());
112  data_.MJtJinv_dIDCdqv().noalias() = data_.MJtJinv() * data_.dIDCdqv();
113  data_.MJtJinv_IDC().noalias() = data_.MJtJinv() * data_.IDC();
114  data_.Qafqv().topRows(dimv_).noalias()
115  = (- kkt_matrix.Qaa().diagonal()).asDiagonal()
116  * data_.MJtJinv_dIDCdqv().topRows(dimv_);
117  data_.Qafqv().bottomRows(dimf).noalias()
118  = - kkt_matrix.Qff() * data_.MJtJinv_dIDCdqv().bottomRows(dimf);
119  data_.Qafu_full().topRows(dimv_).noalias()
120  = kkt_matrix.Qaa().diagonal().asDiagonal()
121  * data_.MJtJinv().topLeftCorner(dimv_, dimv_);
122  data_.Qafu_full().bottomRows(dimf).noalias()
123  = kkt_matrix.Qff() * data_.MJtJinv().bottomLeftCorner(dimf, dimv_);
124  data_.la() = kkt_residual.la;
125  data_.lf() = - kkt_residual.lf();
126  data_.la().noalias()
127  -= kkt_matrix.Qaa().diagonal().asDiagonal()
128  * data_.MJtJinv_IDC().head(dimv_);
129  data_.lf().noalias()
130  -= kkt_matrix.Qff() * data_.MJtJinv_IDC().tail(dimf);
131  kkt_matrix.Qxx().noalias()
132  -= data_.MJtJinv_dIDCdqv().transpose() * data_.Qafqv();
133  kkt_matrix.Qxu_full().noalias()
134  -= data_.MJtJinv_dIDCdqv().transpose() * data_.Qafu_full();
135  kkt_residual.lx().noalias()
136  -= data_.MJtJinv_dIDCdqv().transpose() * data_.laf();
137  kkt_matrix.Quu_full().noalias()
138  += data_.MJtJinv().topRows(dimv_) * data_.Qafu_full();
139  if (has_floating_base_) {
140  kkt_residual.lu_passive.noalias()
141  += data_.MJtJinv().template topRows<kDimFloatingBase>() * data_.laf();
142  }
143  kkt_residual.lu().noalias()
144  += data_.MJtJinv().middleRows(dim_passive_, dimu_) * data_.laf();
145  kkt_matrix.Fvq() = - dt * data_.MJtJinv_dIDCdqv().topLeftCorner(dimv_, dimv_);
146  if (is_forward_euler) {
147  kkt_matrix.Fvv().noalias()
148  = - dt * data_.MJtJinv_dIDCdqv().topRightCorner(dimv_, dimv_)
149  + Eigen::MatrixXd::Identity(dimv_, dimv_);
150  }
151  else {
152  kkt_matrix.Fvv().noalias()
153  = - dt * data_.MJtJinv_dIDCdqv().topRightCorner(dimv_, dimv_)
154  - Eigen::MatrixXd::Identity(dimv_, dimv_);
155  }
156  kkt_matrix.Fvu() = dt * data_.MJtJinv().block(0, dim_passive_, dimv_, dimu_);
157  kkt_residual.Fv().noalias() -= dt * data_.MJtJinv_IDC().head(dimv_);
158 }
