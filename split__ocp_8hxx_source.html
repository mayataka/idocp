251 inline double SplitOCP::squaredNormKKTResidual(
252  const SplitKKTResidual& kkt_residual, const double dt) const {
253  assert(dt > 0);
254  double error = 0;
255  error += kkt_residual.lx().squaredNorm();
256  error += kkt_residual.la.squaredNorm();
257  error += kkt_residual.lf().squaredNorm();
258  if (has_floating_base_) {
259  error += kkt_residual.lu_passive.squaredNorm();
260  }
261  error += kkt_residual.lu().squaredNorm();
262  error += stateequation::squaredNormStateEuqationResidual(kkt_residual);
263  error += contact_dynamics_.squaredNormContactDynamicsResidual(dt);
264  error += dt * dt * constraints_->squaredNormPrimalAndDualResidual(constraints_data_);
265  error += switching_constraint_.squaredNormSwitchingConstraintResidual(kkt_residual);
266  return error;
