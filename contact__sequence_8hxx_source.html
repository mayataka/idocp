63 inline void ContactSequence::push_back(const DiscreteEvent& discrete_event,
64  const double event_time) {
65  try {
66  if (numContactPhases() == 0) {
67  throw std::runtime_error(
68  "Call setContactStatusUniformly() before calling push_back()!");
69  }
70  if (!discrete_event.existDiscreteEvent()) {
71  throw std::runtime_error(
72  "discrete_event.existDiscreteEvent() must be true!");
73  }
74  if (discrete_event.preContactStatus() != contact_statuses_.back()) {
75  throw std::runtime_error(
76  "discrete_event.preContactStatus() is not consistent with the last contact status!");
77  }
78  if (numDiscreteEvents()+1 > max_num_events_) {
79  throw std::runtime_error(
80  "Number of discrete events=" + std::to_string(numDiscreteEvents()+1)
81  + " exceeds predefined max_num_events="
82  + std::to_string(max_num_events_) + "!");
83  }
84  if (numImpulseEvents() > 0 || numLiftEvents() > 0) {
85  if (event_time <= event_time_.back()) {
86  throw std::runtime_error(
87  "event_time=" + std::to_string(event_time)
88  + " must be larger than the last event time="
89  + std::to_string(event_time_.back()) + "!");
90  }
91  }
92  }
93  catch(const std::exception& e) {
94  std::cerr << e.what() << '\n';
95  std::exit(EXIT_FAILURE);
96  }
97  contact_statuses_.push_back(discrete_event.postContactStatus());
98  event_time_.push_back(event_time);
99  if (discrete_event.existImpulse()) {
100  impulse_events_.push_back(discrete_event);
101  event_index_impulse_.push_back(contact_statuses_.size()-2);
102  impulse_time_.push_back(event_time);
103  is_impulse_event_.push_back(true);
104  }
105  else {
106  event_index_lift_.push_back(contact_statuses_.size()-2);
107  lift_time_.push_back(event_time);
108  is_impulse_event_.push_back(false);
109  }
110 }

166 inline void ContactSequence::updateImpulseTime(const int impulse_index,
167  const double impulse_time) {
168  try {
169  if (numImpulseEvents() <= 0) {
170  throw std::runtime_error(
171  "numImpulseEvents() must be positive when calling this method!");
172  }
173  if (impulse_index < 0) {
174  throw std::runtime_error("impulse_index must be non-negative!");
175  }
176  if (impulse_index >= numImpulseEvents()) {
177  throw std::runtime_error(
178  "impulse_index=" + std::to_string(impulse_index)
179  + " must be less than numImpulseEvents()="
180  + std::to_string(numImpulseEvents()) + "!");
181  }
182  const int event_index = event_index_impulse_[impulse_index];
183  if (event_index > 0) {
184  if (event_time_[event_index-1] >= impulse_time) {
185  throw std::runtime_error(
186  "impulse_time=" + std::to_string(impulse_time)
187  + " must be larger than event_time_[event_index-1]="
188  + std::to_string(event_time_[event_index-1]) + "!");
189  }
190  }
191  else if (event_index+1 < numDiscreteEvents()) {
192  if (event_time_[event_index+1] <= impulse_time) {
193  throw std::runtime_error(
194  "impulse_time=" + std::to_string(impulse_time)
195  + " must be larger than event_time_[event_index+1]="
196  + std::to_string(event_time_[event_index+1]) + "!");
197  }
198  }
199  }
200  catch(const std::exception& e) {
201  std::cerr << e.what() << '\n';
202  std::exit(EXIT_FAILURE);
203  }
204  impulse_time_[impulse_index] = impulse_time;
205  event_time_[event_index_impulse_[impulse_index]] = impulse_time;
206 }
