246 inline void OCPDiscretizer::countDiscreteEvents(
247  const ContactSequence& contact_sequence, const double t) {
248  N_impulse_ = contact_sequence.numImpulseEvents();
249  assert(N_impulse_ <= max_events_);
250  for (int i=0; i<N_impulse_; ++i) {
251  t_impulse_[i] = contact_sequence.impulseTime(i);
252  time_stage_before_impulse_[i] = std::floor((t_impulse_[i]-t)/dt_ideal_);
253  }
254  N_lift_ = contact_sequence.numLiftEvents();
255  assert(N_lift_ <= max_events_);
256  for (int i=0; i<N_lift_; ++i) {
257  t_lift_[i] = contact_sequence.liftTime(i);
258  time_stage_before_lift_[i] = std::floor((t_lift_[i]-t)/dt_ideal_);
259  }
260  N_ = N_ideal_;

265 inline void OCPDiscretizer::countTimeSteps(const double t) {
266  int impulse_index = 0;
267  int lift_index = 0;
268  int num_events_on_grid = 0;
269  for (int i=0; i<N_ideal_; ++i) {
270  const int stage = i - num_events_on_grid;
271  if (i == time_stage_before_impulse_[impulse_index]) {
272  dt_[stage] = t_impulse_[impulse_index] - i * dt_ideal_ - t;
273  assert(dt_[stage] >= -min_dt_);
274  assert(dt_[stage] <= dt_ideal_+min_dt_);
275  if (dt_[stage] <= min_dt_) {
276  time_stage_before_impulse_[impulse_index] = stage - 1;
277  dt_aux_[impulse_index] = dt_ideal_;
278  t_[stage] = t + (i-1) * dt_ideal_;
279  ++num_events_on_grid;
280  ++impulse_index;
281  }
282  else if (dt_[stage] >= max_dt_) {
283  time_stage_before_impulse_[impulse_index] = i + 1;
284  t_[stage] = t + i * dt_ideal_;
285  }
286  else {
287  time_stage_before_impulse_[impulse_index] = stage;
288  dt_aux_[impulse_index] = dt_ideal_ - dt_[stage];
289  t_[stage] = t + i * dt_ideal_;
290  ++impulse_index;
291  }
292  }
293  else if (i == time_stage_before_lift_[lift_index]) {
294  dt_[stage] = t_lift_[lift_index] - i * dt_ideal_ - t;
295  assert(dt_[stage] >= -min_dt_);
296  assert(dt_[stage] <= dt_ideal_+min_dt_);
297  if (dt_[stage] <= min_dt_) {
298  time_stage_before_lift_[lift_index] = stage - 1;
299  dt_lift_[lift_index] = dt_ideal_;
300  t_[stage] = t + (i-1) * dt_ideal_;
301  ++num_events_on_grid;
302  ++lift_index;
303  }
304  else if (dt_[stage] >= max_dt_) {
305  time_stage_before_lift_[lift_index] = i + 1;
306  t_[stage] = t + i * dt_ideal_;
307  }
308  else {
309  time_stage_before_lift_[lift_index] = stage;
310  dt_lift_[lift_index] = dt_ideal_ - dt_[stage];
311  t_[stage] = t + i * dt_ideal_;
312  ++lift_index;
313  }
314  }
315  else {
316  dt_[stage] = dt_ideal_;
317  t_[stage] = t + i * dt_ideal_;
318  }
319  }
320  N_ = N_ideal_ - num_events_on_grid;
321  t_[N_] = t + T_;
322 }
