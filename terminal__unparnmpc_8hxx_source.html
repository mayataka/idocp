183 inline double TerminalUnParNMPC::stageCost(Robot& robot, const double t,
184  const double dt,
185  const SplitSolution& s,
186  const double primal_step_size) {
187  assert(dt > 0);
188  assert(primal_step_size >= 0);
189  assert(primal_step_size <= 1);
190  if (use_kinematics_) {
191  robot.updateKinematics(s.q, s.v, s.a);
192  }
193  double cost = 0;
194  cost += cost_->computeStageCost(robot, cost_data_, t, dt, s);
195  cost += cost_->computeTerminalCost(robot, cost_data_, t, s);
196  if (primal_step_size > 0) {
197  cost += dt * constraints_->costSlackBarrier(constraints_data_,
198  primal_step_size);
199  }
200  else {
201  cost += dt * constraints_->costSlackBarrier(constraints_data_);
202  }
203  return cost;
