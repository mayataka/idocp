188 inline void ImpulseSplitSolution::integrate(const Robot& robot,
189  const double step_size,
190  const ImpulseSplitDirection& d) {
191  lmd.noalias() += step_size * d.dlmd();
192  gmm.noalias() += step_size * d.dgmm();
193  robot.integrateConfiguration(d.dq(), step_size, q);
194  v.noalias() += step_size * d.dv();
195  dv.noalias() += step_size * d.ddv();
196  beta.noalias() += step_size * d.dbeta();
197  assert(f_stack().size() == d.df().size());
198  f_stack().noalias() += step_size * d.df();
199  set_f_vector();
200  assert(mu_stack().size() == d.dmu().size());
201  mu_stack().noalias() += step_size * d.dmu();
202  set_mu_vector();
203 }
