40 inline void SplitBackwardCorrection::coarseUpdate(
41  const Robot& robot, const double dt, SplitKKTMatrix& kkt_matrix,
42  const SplitKKTResidual& kkt_residual, const SplitSolution& s,
43  SplitSolution& s_new) {
44  kkt_matrix.Qvq() = kkt_matrix.Qqv().transpose();
45  kkt_matrix.Qux() = kkt_matrix.Qxu().transpose();
46  data_.setImpulseStatus(s.dimi());
47  dimKKT_ = 2*dimx_ + dimu_ + s.dimi();
48  is_switching_constraint_valid_ = s.hasActiveImpulse();
49  if (is_switching_constraint_valid_) {
50  kkt_mat_inverter_.invert(dt, kkt_matrix.Jac(), kkt_matrix.Pq(),
51  kkt_matrix.Qss(), data_.KKT_mat_inv());
52  }
53  else {
54  kkt_mat_inverter_.invert(dt, kkt_matrix.Jac(), kkt_matrix.Qss(),
55  data_.KKT_mat_inv());
56  }
57  data_.splitDirection().noalias()
58  = data_.KKT_mat_inv() * kkt_residual.splitKKTResidual();
59  s_new.lmd = s.lmd - data_.dlmd();
60  s_new.gmm = s.gmm - data_.dgmm();
61  if (is_switching_constraint_valid_) {

63  s_new.xi_stack() = s.xi_stack() - data_.dxi();
64  }
65  s_new.u = s.u - data_.du();
66  robot.integrateConfiguration(s.q, data_.dq(), -1, s_new.q);
67  s_new.v = s.v - data_.dv();
