250 inline double SplitParNMPC::squaredNormKKTResidual(
251  const SplitKKTResidual& kkt_residual, const double dt) const {
252  assert(dt > 0);
253  double error = 0;
254  error += kkt_residual.lx().squaredNorm();
255  error += kkt_residual.la.squaredNorm();

264  error += switchingconstraint::squaredNormSwitchingConstraintResidual(kkt_residual);
265  return error;
