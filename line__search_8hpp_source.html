69  // If filter is empty, augment the current solution to the filter.
70  if (filter_.isEmpty()) {

81  if (filter_.isAccepted(total_costs, total_violations)) {

134  static void computeSolution(const Robot& robot, const SplitSolution& s,
135  const SplitDirection& d, const double step_size,
136  SplitSolution& s_try) {
137  s_try.setContactStatus(s);
138  robot.integrateConfiguration(s.q, d.dq(), step_size, s_try.q);
139  s_try.v = s.v + step_size * d.dv();
140  s_try.a = s.a + step_size * d.da();
141  s_try.u = s.u + step_size * d.du();
142  if (s.hasActiveContacts()) {
143  s_try.f_stack() = s.f_stack() + step_size * d.df();
