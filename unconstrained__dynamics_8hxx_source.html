55 inline void UnconstrainedDynamics::linearizeUnconstrainedDynamics(
56  Robot& robot, const double dt, const SplitSolution& s,
57  SplitKKTResidual& kkt_residual) {
58  assert(dt > 0);
59  linearizeInverseDynamics(robot, s);
60  // augment inverse dynamics constraint
61  kkt_residual.lq().noalias() += dt * dID_dq_.transpose() * s.beta;
62  kkt_residual.lv().noalias() += dt * dID_dv_.transpose() * s.beta;
63  kkt_residual.la.noalias() += dt * dID_da_.transpose() * s.beta;
64  kkt_residual.lu().noalias() -= dt * s.beta;
65 }

97 inline void UnconstrainedDynamics::computeCondensedDirection(
98  const double dt, const SplitKKTMatrix& kkt_matrix,
99  const SplitKKTResidual& kkt_residual, SplitDirection& d) {
100  assert(dt > 0);
101  d.du() = ID_;
102  d.du().noalias() += dID_dq_ * d.dq();
103  d.du().noalias() += dID_dv_ * d.dv();
104  d.du().noalias() += dID_da_ * d.da();
