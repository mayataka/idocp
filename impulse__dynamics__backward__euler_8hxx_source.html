29 inline void ImpulseDynamicsBackwardEuler::linearizeImpulseDynamics(
30  Robot& robot, const ImpulseStatus& impulse_status,
31  const ImpulseSplitSolution& s, ImpulseSplitKKTMatrix& kkt_matrix,
32  ImpulseSplitKKTResidual& kkt_residual) {
33  setImpulseStatus(impulse_status);
34  linearizeInverseImpulseDynamics(robot, impulse_status, s, data_);
35  linearizeImpulseVelocityConstraint(robot, impulse_status, kkt_matrix,
36  kkt_residual);
37  // augment inverse impulse dynamics constraint
38  kkt_residual.lq().noalias() += data_.dImDdq.transpose() * s.beta;
39  kkt_residual.ldv.noalias() += data_.dImDddv.transpose() * s.beta;
40  // We use an equivalence dmIDdf().transpose() = - Vv() to avoid
41  // redundant calculation of dImDdf().
42  kkt_residual.lf().noalias() -= kkt_matrix.Vv() * s.beta;
43  // augment impulse velocity constraint
44  kkt_residual.lq().noalias() += kkt_matrix.Vq().transpose() * s.mu_stack();
45  kkt_residual.lv().noalias() += kkt_matrix.Vv().transpose() * s.mu_stack();
46 }

75 inline void ImpulseDynamicsBackwardEuler::condensing(
76  const Robot& robot, ImpulseDynamicsBackwardEulerData& data,
77  ImpulseSplitKKTMatrix& kkt_matrix, ImpulseSplitKKTResidual& kkt_residual) {
78  const int dimv = robot.dimv();
79  kkt_matrix.Fvq().noalias() = - data.Minv * data.dImDdq; // this is - Minv_dImDdq
80  kkt_matrix.Fvf().noalias() = data.Minv * kkt_matrix.Vv().transpose(); // this is - Minv_dImDdf
81  data.Minv_ImD.noalias() = data.Minv * data.ImD;
82  data.Qdvq.noalias()
83  = kkt_matrix.Qdvdv().diagonal().asDiagonal() * kkt_matrix.Fvq();
84  data.Qdvf().noalias()
85  = kkt_matrix.Qdvdv().diagonal().asDiagonal() * kkt_matrix.Fvf();
86  data.ldv = kkt_residual.ldv;
87  data.ldv.noalias()
88  -= kkt_matrix.Qdvdv().diagonal().asDiagonal() * data.Minv_ImD;
89  kkt_matrix.Qqq().noalias() += kkt_matrix.Fvq().transpose() * data.Qdvq;
90  kkt_matrix.Qfq().transpose().noalias()
91  += kkt_matrix.Fvq().transpose() * data.Qdvf();
92  kkt_matrix.Qff().noalias() += kkt_matrix.Fvf().transpose() * data.Qdvf();
93  kkt_residual.lq().noalias() += kkt_matrix.Fvq().transpose() * data.ldv;
94  kkt_residual.lf().noalias() += kkt_matrix.Fvf().transpose() * data.ldv;
95  kkt_matrix.Fvv() = - Eigen::MatrixXd::Identity(dimv, dimv);
96  kkt_residual.Fv().noalias() -= data.Minv_ImD;
97 }
