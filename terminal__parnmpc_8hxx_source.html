170 inline double TerminalParNMPC::squaredNormKKTResidual(
171  const SplitKKTResidual& kkt_residual, const double dt) const {
172  assert(dt > 0);
173  double error = 0;
174  error += kkt_residual.lx().squaredNorm();
175  error += kkt_residual.la.squaredNorm();
176  error += kkt_residual.lf().squaredNorm();
177  if (has_floating_base_) {
178  error += kkt_residual.lu_passive.squaredNorm();
179  }
180  error += kkt_residual.lu().squaredNorm();
181  error += stateequation::squaredNormStateEuqationResidual(kkt_residual);
182  error += contact_dynamics_.squaredNormContactDynamicsResidual(dt);
183  error += constraints_->squaredNormPrimalAndDualResidual(constraints_data_);
184  return error;

188 inline double TerminalParNMPC::stageCost(Robot& robot, const double t,
189  const double dt,
190  const SplitSolution& s,
191  const double primal_step_size) {
192  assert(dt > 0);
193  assert(primal_step_size >= 0);
194  assert(primal_step_size <= 1);
195  if (use_kinematics_) {
196  robot.updateKinematics(s.q, s.v, s.a);
197  }
198  double cost = 0;
199  cost += cost_->computeStageCost(robot, cost_data_, t, dt, s);
200  if (primal_step_size > 0) {
201  cost += dt * constraints_->costSlackBarrier(constraints_data_, primal_step_size);
202  }
203  else {
204  cost += dt * constraints_->costSlackBarrier(constraints_data_);
205  }
206  return cost;
207 }
