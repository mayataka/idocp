49  const double dt, SplitUnKKTMatrix& unkkt_matrix,
50  const SplitUnKKTResidual& unkkt_residual, const SplitSolution& s,
51  SplitDirection& d, SplitSolution& s_new) {
52  unkkt_matrix.Qvq() = unkkt_matrix.Qqv().transpose();
53  unkkt_matrix.Qxa() = unkkt_matrix.Qax().transpose();
