55 inline void SplitRiccatiFactorizer::backwardRiccatiRecursion(
56  const SplitRiccatiFactorization& riccati_next, const double dt,
57  SplitKKTMatrix& kkt_matrix, SplitKKTResidual& kkt_residual,
58  const SplitStateConstraintJacobian& jac, SplitRiccatiFactorization& riccati,
59  SplitConstrainedRiccatiFactorization& c_riccati) {
60  assert(dt > 0);
61  backward_recursion_.factorizeKKTMatrix(riccati_next, dt, kkt_matrix,
62  kkt_residual);
63  // Schur complement
64  llt_.compute(kkt_matrix.Quu());
65  assert(llt_.info() == Eigen::Success);
66  c_riccati.setImpulseStatus(jac.dimi());
67  c_riccati.Ginv.noalias() = llt_.solve(Eigen::MatrixXd::Identity(dimu_, dimu_));
68  c_riccati.DGinv().transpose().noalias() = llt_.solve(jac.Phiu().transpose());
69  c_riccati.S().noalias() = c_riccati.DGinv() * jac.Phiu().transpose();
70  llt_s_.compute(c_riccati.S());
71  assert(llt_s_.info() == Eigen::Success);
72  c_riccati.SinvDGinv().noalias() = llt_s_.solve(c_riccati.DGinv());
73  c_riccati.Ginv.noalias() -= c_riccati.SinvDGinv().transpose() * c_riccati.DGinv();
74  lqr_policy_.K.noalias() = - c_riccati.Ginv * kkt_matrix.Qxu().transpose();
75  lqr_policy_.K.noalias() -= c_riccati.SinvDGinv().transpose() * jac.Phix();
76  lqr_policy_.k.noalias() = - c_riccati.Ginv * kkt_residual.lu();
77  lqr_policy_.k.noalias() -= c_riccati.SinvDGinv().transpose() * kkt_residual.P();
78  c_riccati.M().noalias() = llt_s_.solve(jac.Phix());
79  c_riccati.M().noalias() -= c_riccati.SinvDGinv() * kkt_matrix.Qxu().transpose();
80  c_riccati.m().noalias() = llt_s_.solve(kkt_residual.P());
81  c_riccati.m().noalias() -= c_riccati.SinvDGinv() * kkt_residual.lu();
82  assert(!lqr_policy_.K.hasNaN());
83  assert(!lqr_policy_.k.hasNaN());
84  assert(!c_riccati.M().hasNaN());
85  assert(!c_riccati.m().hasNaN());
86  backward_recursion_.factorizeRiccatiFactorization(riccati_next, kkt_matrix,
87  kkt_residual, lqr_policy_,
88  dt, riccati);
89  c_riccati.DtM.noalias() = jac.Phiu().transpose() * c_riccati.M();
90  c_riccati.KtDtM.noalias() = lqr_policy_.K.transpose() * c_riccati.DtM;
91  riccati.Pqq.noalias() -= c_riccati.KtDtM.topLeftCorner(dimv_, dimv_);
92  riccati.Pqq.noalias() -= c_riccati.KtDtM.topLeftCorner(dimv_, dimv_).transpose();
93  riccati.Pqv.noalias() -= c_riccati.KtDtM.topRightCorner(dimv_, dimv_);
94  riccati.Pqv.noalias() -= c_riccati.KtDtM.bottomLeftCorner(dimv_, dimv_).transpose();
95  riccati.Pvq = riccati.Pqv.transpose();
96  riccati.Pvv.noalias() -= c_riccati.KtDtM.bottomRightCorner(dimv_, dimv_);
97  riccati.Pvv.noalias() -= c_riccati.KtDtM.bottomRightCorner(dimv_, dimv_).transpose();
98  riccati.sq.noalias() -= jac.Phix().transpose().topRows(dimv_) * c_riccati.m();
99  riccati.sv.noalias() -= jac.Phix().transpose().bottomRows(dimv_) * c_riccati.m();
100 }

103 template <typename SplitDirectionType>
104 inline void SplitRiccatiFactorizer::forwardRiccatiRecursion(
105  const SplitKKTMatrix& kkt_matrix, const SplitKKTResidual& kkt_residual,
106  const double dt, SplitDirection& d, SplitDirectionType& d_next) const {
107  assert(dt > 0);
108  d.du().noalias() = lqr_policy_.K * d.dx();
109  d.du().noalias() += lqr_policy_.k;
110  d_next.dx() = kkt_residual.Fx();
111  if (has_floating_base_) {
112  d_next.dq().template head<6>().noalias()
113  += kkt_matrix.Fqq().template topLeftCorner<6, 6>()
114  * d.dq().template head<6>();
115  d_next.dq().tail(dimv_-6).noalias() += d.dq().tail(dimv_-6);
116  d_next.dq().template head<6>().noalias()
117  += kkt_matrix.Fqv().template topLeftCorner<6, 6>()
118  * d.dv().template head<6>();
119  d_next.dq().tail(dimv_-6).noalias() += dt * d.dv().tail(dimv_-6);
120  }
121  else {
122  d_next.dq().noalias() += d.dq();
123  d_next.dq().noalias() += dt * d.dv();
124  }
125  d_next.dv().noalias() += kkt_matrix.Fvq() * d.dq();
126  d_next.dv().noalias() += kkt_matrix.Fvv() * d.dv();
127  d_next.dv().noalias() += kkt_matrix.Fvu() * d.du();
128 }
