268 inline void ParNMPCDiscretizer::countDiscreteEvents(
269  const ContactSequence& contact_sequence, const double t) {
270  N_impulse_ = contact_sequence.numImpulseEvents();
271  assert(N_impulse_ <= max_events_);
272  for (int i=0; i<N_impulse_; ++i) {
273  t_impulse_[i] = contact_sequence.impulseTime(i);
274  time_stage_after_impulse_[i] = std::floor((t_impulse_[i]-t)/dt_ideal_);
275  }
276  N_lift_ = contact_sequence.numLiftEvents();
277  assert(N_lift_ <= max_events_);
278  for (int i=0; i<N_lift_; ++i) {
279  t_lift_[i] = contact_sequence.liftTime(i);
280  time_stage_after_lift_[i] = std::floor((t_lift_[i]-t)/dt_ideal_);
281  }
282  N_ = N_ideal_;
283  assert(isWellDefined());

287 inline void ParNMPCDiscretizer::countTimeSteps(const double t) {
288  int impulse_index = 0;
289  int lift_index = 0;
290  int num_events_on_grid = 0;
291  for (int i=0; i<N_ideal_; ++i) {
292  const int stage = i - num_events_on_grid;
293  if (i == time_stage_after_impulse_[impulse_index]) {
294  dt_[stage] = (i+1) * dt_ideal_ + t - t_impulse_[impulse_index];
295  assert(dt_[stage] >= -min_dt_);
296  assert(dt_[stage] <= dt_ideal_+min_dt_);
297  if (dt_[stage] <= min_dt_) {
298  time_stage_after_impulse_[impulse_index] = i + 1;
299  t_[stage] = t + (i+1) * dt_ideal_;
300  }
301  else if (dt_[stage] >= max_dt_) {
302  time_stage_after_impulse_[impulse_index] = stage - 1;
303  dt_aux_[impulse_index] = dt_ideal_;
304  t_[stage] = t + i * dt_ideal_;
305  ++num_events_on_grid;
306  ++impulse_index;
307  }
308  else {
309  time_stage_after_impulse_[impulse_index] = stage;
310  dt_aux_[impulse_index] = dt_ideal_ - dt_[stage];
311  t_[stage] = t + (i+1) * dt_ideal_;
312  ++impulse_index;
313  }
314  }
315  else if (i == time_stage_after_lift_[lift_index]) {
316  dt_[stage] = (i+1) * dt_ideal_ + t - t_lift_[lift_index];
317  assert(dt_[stage] >= -min_dt_);
318  assert(dt_[stage] <= dt_ideal_+min_dt_);
319  if (dt_[stage] <= min_dt_) {
320  time_stage_after_lift_[lift_index] = i + 1;
321  t_[stage] = t + (i+1) * dt_ideal_;
322  }
323  else if (dt_[stage] >= max_dt_) {
324  time_stage_after_lift_[lift_index] = stage - 1;
325  dt_lift_[lift_index] = dt_ideal_;
326  t_[stage] = t + i * dt_ideal_;
327  ++num_events_on_grid;
328  ++lift_index;
329  }
330  else {
331  time_stage_after_lift_[lift_index] = stage;
332  dt_lift_[lift_index] = dt_ideal_ - dt_[stage];
333  t_[stage] = t + (i+1) * dt_ideal_;
334  ++lift_index;
335  }
336  }
337  else {
338  dt_[stage] = dt_ideal_;
339  t_[stage] = t + (i+1) * dt_ideal_;
340  }
341  }
342  N_ = N_ideal_ - num_events_on_grid;
343  t_[N_-1] = t + T_;
344 }
