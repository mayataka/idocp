68  AtPqv_ = riccati_next.Pqv;

117 inline void BackwardRiccatiRecursionFactorizer::factorizeRiccatiFactorization(
118  const SplitRiccatiFactorization& riccati_next,
119  const SplitKKTMatrix& kkt_matrix, const SplitKKTResidual& kkt_residual,
120  const LQRStateFeedbackPolicy& lqr_policy, const double dt,
121  SplitRiccatiFactorization& riccati) {
122  assert(dt > 0);
123  riccati.Pqq = kkt_matrix.Qqq();
124  riccati.Pqv = kkt_matrix.Qqv();
125  riccati.Pvv = kkt_matrix.Qvv();
126  GK_.noalias() = kkt_matrix.Quu() * lqr_policy.K;
127  riccati.Pqq.noalias()
128  -= lqr_policy.K.leftCols(dimv_).transpose() * GK_.leftCols(dimv_);
129  riccati.Pqv.noalias()
130  -= lqr_policy.K.leftCols(dimv_).transpose() * GK_.rightCols(dimv_);
131  riccati.Pvv.noalias()
132  -= lqr_policy.K.rightCols(dimv_).transpose() * GK_.rightCols(dimv_);
133  riccati.Pvq = riccati.Pqv.transpose();
134  // preserve the symmetry
135  riccati.Pqq = 0.5 * (riccati.Pqq + riccati.Pqq.transpose()).eval();
136  riccati.Pvv = 0.5 * (riccati.Pvv + riccati.Pvv.transpose()).eval();
137  if (has_floating_base_) {
138  riccati.sq.template head<6>().noalias()
139  = kkt_matrix.Fqq().template topLeftCorner<6, 6>().transpose()
140  * riccati_next.sq.template head<6>();
141  riccati.sq.tail(dimv_-6) = riccati_next.sq.tail(dimv_-6);
142  riccati.sv.template head<6>().noalias()
143  = kkt_matrix.Fqv().template topLeftCorner<6, 6>().transpose()
144  * riccati_next.sq.template head<6>();
145  riccati.sv.tail(dimv_-6) = dt * riccati_next.sq.tail(dimv_-6);
146  }
147  else {
148  riccati.sq.noalias() = riccati_next.sq;
149  riccati.sv.noalias() = dt * riccati_next.sq;
150  }
151  riccati.sq.noalias() += kkt_matrix.Fvq().transpose() * riccati_next.sv;
152  riccati.sv.noalias() += kkt_matrix.Fvv().transpose() * riccati_next.sv;
153  riccati.sq.noalias() -= AtPqq_ * kkt_residual.Fq();
154  riccati.sq.noalias() -= AtPqv_ * kkt_residual.Fv();
155  riccati.sv.noalias() -= AtPvq_ * kkt_residual.Fq();
156  riccati.sv.noalias() -= AtPvv_ * kkt_residual.Fv();
157  riccati.sq.noalias() -= kkt_residual.lq();
158  riccati.sv.noalias() -= kkt_residual.lv();
159  riccati.sq.noalias() -= kkt_matrix.Qqu() * lqr_policy.k;
160  riccati.sv.noalias() -= kkt_matrix.Qvu() * lqr_policy.k;
161 }
