145  const ImpulseSplitKKTResidual& kkt_residual) const {
146  double error = 0;

157 inline double ImpulseSplitParNMPC::stageCost(Robot& robot, const double t,
158  const ImpulseSplitSolution& s,
159  const double primal_step_size) {
160  assert(primal_step_size >= 0);
161  assert(primal_step_size <= 1);
162  robot.updateKinematics(s.q, s.v);
163  double cost = 0;
164  cost += cost_->computeImpulseCost(robot, cost_data_, t, s);
165  if (primal_step_size > 0) {
166  cost += constraints_->costSlackBarrier(constraints_data_, primal_step_size);
167  }
168  else {
169  cost += constraints_->costSlackBarrier(constraints_data_);
170  }
171  return cost;
