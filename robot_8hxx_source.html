575 template <typename MatrixType1, typename MatrixType2, typename MatrixType3>
576 inline void Robot::computeMJtJinv(
577  const Eigen::MatrixBase<MatrixType1>& M,
578  const Eigen::MatrixBase<MatrixType2>& J,
579  const Eigen::MatrixBase<MatrixType3>& MJtJinv) {
580  assert(M.rows() == dimv_);
581  assert(M.cols() == dimv_);
582  assert(J.rows() <= max_dimf_);
583  assert(J.cols() == dimv_);
584  assert(MJtJinv.rows() == M.rows()+J.rows());
585  assert(MJtJinv.cols() == M.rows()+J.rows());
586  const int dimf = J.rows();
587  data_.M = M;
588  pinocchio::cholesky::decompose(model_, data_);
589  data_.sDUiJt.leftCols(dimf) = J.transpose();
590  pinocchio::cholesky::Uiv(model_, data_, data_.sDUiJt.leftCols(dimf));
591  for (Eigen::DenseIndex k=0; k<dimv_; ++k) {
592  data_.sDUiJt.leftCols(dimf).row(k) /= std::sqrt(data_.D[k]);
593  }
594  data_.JMinvJt.topLeftCorner(dimf, dimf).noalias()
595  = data_.sDUiJt.leftCols(dimf).transpose() * data_.sDUiJt.leftCols(dimf);
596  data_.llt_JMinvJt.compute(data_.JMinvJt.topLeftCorner(dimf, dimf));
597  assert(data_.llt_JMinvJt.info() == Eigen::Success);
598  Eigen::Block<MatrixType3> topLeft
599  = const_cast<Eigen::MatrixBase<MatrixType3>&>(MJtJinv).topLeftCorner(dimv_, dimv_);
600  Eigen::Block<MatrixType3> topRight
601  = const_cast<Eigen::MatrixBase<MatrixType3>&>(MJtJinv).topRightCorner(dimv_, dimf);
602  Eigen::Block<MatrixType3> bottomLeft
603  = const_cast<Eigen::MatrixBase<MatrixType3>&>(MJtJinv).bottomLeftCorner(dimf, dimv_);
604  Eigen::Block<MatrixType3> bottomRight
605  = const_cast<Eigen::MatrixBase<MatrixType3>&>(MJtJinv).bottomRightCorner(dimf, dimf);
606  bottomRight = - pinocchio::Data::MatrixXs::Identity(dimf, dimf);
607  topLeft.setIdentity();
608  data_.llt_JMinvJt.solveInPlace(bottomRight);
609  pinocchio::cholesky::solve(model_, data_, topLeft);
610  bottomLeft.noalias() = J * topLeft;
611  topRight.noalias() = bottomLeft.transpose() * (-bottomRight);
612  topLeft.noalias() -= topRight*bottomLeft;
613  bottomLeft = topRight.transpose();
614  assert(!MJtJinv.hasNaN());
615 }
