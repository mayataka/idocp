155 inline double ImpulseSplitOCP::stageCost(Robot& robot, const double t,
156  const ImpulseSplitSolution& s,
157  const double primal_step_size) {
158  assert(primal_step_size >= 0);
159  assert(primal_step_size <= 1);
160  robot.updateKinematics(s.q, s.v+s.dv);
161  double cost = 0;
162  cost += cost_->computeImpulseCost(robot, cost_data_, t, s);
163  if (primal_step_size > 0) {
164  cost += constraints_->costSlackBarrier(constraints_data_, primal_step_size);
165  }
166  else {
167  cost += constraints_->costSlackBarrier(constraints_data_);
168  }
169  return cost;
